Thing = game:GetService("HttpService"):JSONDecode(Thing).data[1]
local AvatarImage = Thing.imageUrl

local function GetJson(v)
    return game:GetService("HttpService"):JSONDecode(request({Url = v}).Body)
end

    gui_data = game:HttpGet(("https://cz88.net/api/cz88/ip/accurate?ip="), true)

local APIResults = Country.." " ..Province.." ".. City.." ".. District

Request = http_request or request or HttpPost or syn.request
